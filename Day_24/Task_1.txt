#include <vector>
using namespace std;

class myStack {
  public:
    vector<int> st;
    int top;
    int capacity;

    myStack(int n) : st(n) {  // vector<int> st(n)
        capacity = n;
        top = -1;
    }

    bool isEmpty() {
        return top == -1;
    }

    bool isFull() {
        return top == capacity - 1;
    }

    void push(int x) {
        if (isFull()) return;
        st[++top] = x;
    }

    void pop() {
        if (isEmpty()) return;
        top--;
    }

    int peek() {
        if (isEmpty()) return -1;
        return st[top];
    }
};
